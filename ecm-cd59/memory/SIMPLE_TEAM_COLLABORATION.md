### 2. 智能体间协作方式

#### 任务分配与执行
```javascript
// 智能体间协作伪代码
function assignTask(task, agent) {
    const agentStatus = checkAgentAvailability(agent);
    if (agentStatus.available) {
        sendTaskToAgent(task, agent);
        trackTaskProgress(task);
    } else {
        findAlternativeAgent(task);
    }
}

function reviewWork(work, reviewer) {
    const reviewResult = evaluateWorkQuality(work);
    if (reviewResult.pass) {
        markWorkAsComplete(work);
        notifyTeamSuccess(work);
    } else {
        requestRevisions(work, reviewResult.feedback);
    }
}
```
