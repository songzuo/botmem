// 核心数据结构
interface AgentCapability {
  agentId: string
  name: string
  provider: 'minimax' | 'bailian' | 'volcengine' | 'self-claude'
  capabilities: {
    techStack: string[] // 技术栈
    taskTypes: TaskType[] // 任务类型
    concurrency: number // 并发能力
    avgResponseTime: number // 平均响应时间
    successRate: number // 成功率
  }
  currentLoad: number // 当前负载 (0-100)
  availability: boolean // 可用性
  lastActiveTime: number // 最后活跃时间
}

interface Task {
  id: string
  type: TaskType
  priority: 'low' | 'medium' | 'high' | 'urgent'
  deadline?: number
  requiredCapabilities: string[]
  estimatedDuration: number
  dependencies: string[] // 依赖的任务 ID
  status: 'pending' | 'assigned' | 'in_progress' | 'completed' | 'failed'
  assignedAgent?: string
  createdAt: number
}

interface ScheduleDecision {
  taskId: string
  assignedAgent: string
  confidence: number // 分配置信度 (0-1)
  reasoning: string // 分配原因
  alternativeAgents: string[] // 备选 Agent
  estimatedCompletion: number // 预计完成时间
}

// 调度算法核心
class AgentScheduler {
  private agents: Map<string, AgentCapability>
  private tasks: Map<string, Task>
  private decisionHistory: ScheduleDecision[]

  // 1. 任务匹配
  async scheduleTask(task: Task): Promise<ScheduleDecision> {
    const candidates = this.findCandidates(task)
    const ranked = this.rankCandidates(task, candidates)
    const decision = this.makeDecision(task, ranked)
    await this.assignTask(task, decision)
    return decision
  }

  // 2. 寻找候选 Agent
  private findCandidates(task: Task): AgentCapability[] {
    return Array.from(this.agents.values()).filter(
      agent =>
        agent.availability &&
        agent.capabilities.taskTypes.includes(task.type) &&
        this.hasRequiredCapabilities(agent, task.requiredCapabilities) &&
        this.checkLoadCapacity(agent, task)
    )
  }

  // 3. 候选排序
  private rankCandidates(task: Task, candidates: AgentCapability[]): AgentCapability[] {
    return candidates.sort((a, b) => {
      const scoreA = this.calculateScore(task, a)
      const scoreB = this.calculateScore(task, b)
      return scoreB - scoreA
    })
  }

  // 4. 评分算法
  private calculateScore(task: Task, agent: AgentCapability): number {
    const capabilityScore = this.matchCapabilityScore(task, agent)
    const loadScore = 100 - agent.currentLoad
    const performanceScore = agent.successRate * 100
    const responseScore = 100 - Math.min(agent.avgResponseTime / 10, 100)

    // 权重配置
    const weights = {
      capability: 0.4,
      load: 0.3,
      performance: 0.2,
      response: 0.1,
    }

    return (
      capabilityScore * weights.capability +
      loadScore * weights.load +
      performanceScore * weights.performance +
      responseScore * weights.response
    )
  }

  // 5. 负载检查
  private checkLoadCapacity(agent: AgentCapability, task: Task): boolean {
    const estimatedLoad = task.estimatedDuration / 60 // 转换为分钟负载
    return agent.currentLoad + estimatedLoad < 90 // 保留 10% 缓冲
  }
}

// 异常检测核心
class PerformanceAnomalyDetector {
  private baselines: Map<string, MetricBaseline>;
  private history: MetricHistory[];

  // 1. 检测异常
  detectAnomaly(metric: string, value: number): AnomalyDetection {
    const baseline = this.baselines.get(metric);
    if (!baseline) {
      return { isAnomaly: false, reason: 'No baseline' };
    }

    // Z-score 检测
    const zScore = Math.abs((value - baseline.mean) / baseline.stdDev);
    if (zScore > 3) {
      return {
        isAnomaly: true,
        severity: 'critical',
        zScore,
        confidence: this.calculateConfidence(zScore),
        reason: `Z-score: ${zScore.toFixed(2)}`
      };
    }

    // 百分比检测
    const percentChange = ((value - baseline.mean) / baseline.mean) * 100;
    if (Math.abs(percentChange) > 50) {
      return {
        isAnomaly: true,
        severity: 'warning',
        percentChange,
        confidence: this.calculateConfidence(percentChange),
        reason: `Change: ${percentChange.toFixed(1)}%`
      };
    }

    return { isAnomaly: false };
  }

  // 2. 根因分析
  async analyzeRootCause(metric: Metric, context: PerformanceContext): Promise<RootCause> {
    const causes: RootCauseCandidate[] = [];

    // 检查数据库查询
    if (context.slowQueries?.length > 0) {
      causes.push({
        type: 'database',
        severity: 'high',
        description: 'Slow database queries detected',
        details: context.slowQueries.map(q => ({
          query: q.query,
          duration: q.duration,
          affected: q.rowCount
        }))
      });
    }

    // 检查 API 调用
    if (context.slowApis?.length > 0) {
      causes.push({
        type: 'api',
        severity: 'medium',
        description: 'Slow API calls detected',
        details: context.slowApis
      });
    }

    // 检查渲染性能
    if (context.rendering?.longTasks > 10) {
      causes.push({
        type: 'rendering',
        severity: 'medium',
        description: 'Long rendering tasks detected',
        details: {
          longTaskCount: context.rendering.longTasks,
          totalBlockingTime: context.rendering.totalBlockingTime
        }
      });
    }

    // 排序并返回最可能的根因
    return causes.sort((a, b) => this.severityScore(b.severity) - this.severityScore(a.severity))[0];
  }

  // 3. 性能预算检查
  checkBudget(page: string, metrics: PerformanceMetrics): BudgetCheckResult {
    const budget = this.getBudget(page);
    const violations: BudgetViolation[] = [];

    for (const entry of budget.timings) {
      const actual = metrics[entry.metric];
      const threshold = entry.budget * (1 + entry.tolerance);

      if (actual > threshold) {
        violations.push({
          metric: entry.metric,
          budget: entry.budget,
          actual,
          threshold,
          percentOver: ((actual - threshold) / threshold) * 100
        });
      }
    }

    return {
      passed: violations.length === 0,
      violations
    };
  }
}

class PerformanceAlerter {

  async sendAlert(alert: PerformanceAlert): Promise<void> {
    // 告警抑制（避免重复告警）
    if (await this.shouldSuppress(alert)) {

    // 发送到所有配置的渠道
    await Promise.all(
      this.channels.map(channel => channel.send(aggregated))
    );
  }
}

      throw new Error('Room is private')

        return false
      }

    if (!room.members.find(m => m.userId === userId)) {
      room.members.push({
        userId,
        role: 'member',
        permissions: this.getDefaultPermissions('member'),
        joinedAt: Date.now(),
      })
      await this.persistRoom(room)
    }

    return true
  }

    if (!room) {
      return false
    }

    const member = room.members.find(m => m.userId === userId)
    if (!member) {
      return false
    }

    roomId: string,
    options: {
      limit?: number
      before?: number
      after?: number
    }
  ): Promise<Message[]> {
    let query = this.db.messages.where('roomId').equals(roomId)

    if (options.before) {
      query = query.filter(m => m.createdAt < options.before)
    }

    if (options.after) {
      query = query.filter(m => m.createdAt > options.after)
    }

    if (options.limit) {
      query = query.limit(options.limit)
    }

    return query.reverse().toArray()
  }

  async markAsRead(messageId: string, userId: string): Promise<void> {
    await this.db.messages.update(messageId, {
      readBy: this.db.messages.get(messageId).then(msg => [...(msg?.readBy || []), userId]),
    })
  }

  async searchMessages(query: string, roomId?: string): Promise<Message[]> {
    return this.db.messages
      .filter(m => {
        const roomMatch = !roomId || m.roomId === roomId
        const contentMatch =
          m.type === 'text' && m.content.toLowerCase().includes(query.toLowerCase())
        return roomMatch && contentMatch && !m.deletedAt
      })
      .toArray()
  }
}

class ReactCompilerDiagnostics {
  // 扫描不兼容的组件
  async scanIncompatibleComponents(): Promise<IncompatibilityReport[]> {
    const components = await this.getAllComponents();
    const reports: IncompatibilityReport[] = [];

    for (const component of components) {
      const issues = await this.checkComponent(component);
      if (issues.length > 0) {
        reports.push({
          component: component.path,
          issues,
          severity: this.calculateSeverity(issues)
        });
      }
    }

    return reports;
  }

  // 检查单个组件
  private async checkComponent(component: ComponentInfo): Promise<CompilerIssue[]> {
    const issues: CompilerIssue[] = [];
    const code = await component.readCode();

    // 检查 1: 使用了 ref 的 current 属性（编译器不支持）
    if (/\bref\.current\b/.test(code)) {
      issues.push({
        type: 'unsupported-pattern',
        message: 'Using ref.current is not supported by React Compiler',
        line: this.findLineNumber(code, 'ref.current'),
        suggestion: 'Use useRef hook or state instead'
      });
    }

    // 检查 2: 使用了 dangerouslySetInnerHTML
    if (/\bdangerouslySetInnerHTML\b/.test(code)) {
      issues.push({
        type: 'unsupported-pattern',
        message: 'Using dangerouslySetInnerHTML is not supported',
        line: this.findLineNumber(code, 'dangerouslySetInnerHTML'),
        suggestion: 'Use a safe HTML sanitization library'
      });
    }

    // 检查 3: 使用了第三方库的副作用
    if (this.hasThirdPartySideEffects(code)) {
      issues.push({
        type: 'side-effect',
        message: 'Component has potential side effects from third-party library',
        line: this.findSideEffectLine(code),
        suggestion: 'Use React.memo or useMemo to optimize'
      });
    }

    return issues;
  }

  // 性能对比
  async comparePerformance(component: string): Promise<PerformanceComparison> {
    const withoutCompiler = await this.measureComponentWithoutCompiler(component);
    const withCompiler = await this.measureComponentWithCompiler(component);

    return {
      component,
      renders: {
        without: withoutCompiler.renders,
        with: withCompiler.renders,
        improvement: ((withoutCompiler.renders - withCompiler.renders) / withoutCompiler.renders) * 100
      },
      renderTime: {
        without: withoutCompiler.avgRenderTime,
        with: withCompiler.avgRenderTime,
        improvement: ((withoutCompiler.avgRenderTime - withCompiler.avgRenderTime) / withoutCompiler.avgRenderTime) * 100
      },
      memory: {
        without: withoutCompiler.memory,
        with: withCompiler.memory,
        improvement: ((withoutCompiler.memory - withCompiler.memory) / withoutCompiler.memory) * 100
      }
    };
  }
}

// src/lib/prefetch/predictive-prefetcher.ts
class PredictivePrefetcher {
  private userBehavior: Map<string, UserBehaviorPattern>
  private prefetchCache: Map<string, PrefetchResult>

  // 预测下一个可能访问的页面
  predictNextPages(currentPath: string, userContext: UserContext): string[] {
    const predictions: { path: string; confidence: number }[] = []

    // 基于历史模式预测
    const pattern = this.userBehavior.get(currentPath)
    if (pattern) {
      for (const nextPath of pattern.nextPages) {
        predictions.push({
          path: nextPath.path,
          confidence: nextPath.frequency / pattern.totalVisits,
        })
      }
    }

    // 基于用户上下文预测
    if (userContext.taskContext?.type === 'task-editing') {
      predictions.push({
        path: '/tasks',
        confidence: 0.7,
      })
    }

    // 按置信度排序并返回前 3 个
    return predictions
      .sort((a, b) => b.confidence - a.confidence)
      .slice(0, 3)
      .map(p => p.path)
  }

  // 预加载页面
  async prefetchPages(paths: string[]): Promise<void> {
    for (const path of paths) {
      if (!this.prefetchCache.has(path)) {
        const result = await this.prefetchPage(path)
        this.prefetchCache.set(path, result)
      }
    }
  }

  private async prefetchPage(path: string): Promise<PrefetchResult> {
    const startTime = performance.now()
    try {
      // 预加载路由数据
      const data = await this.prefetchRouteData(path)
      // 预加载关键资源
      await this.prefetchCriticalResources(path)

      return {
        path,
        success: true,
        duration: performance.now() - startTime,
        timestamp: Date.now(),
      }
    } catch (error) {
      return {
        path,
        success: false,
        error: error.message,
        duration: performance.now() - startTime,
        timestamp: Date.now(),
      }
    }
  }
}

class ShortcutManager {
  private shortcuts: Map<string, KeyboardShortcut>
  private context: ShortcutContext = 'global'

  registerShortcut(shortcut: KeyboardShortcut): void {
    this.shortcuts.set(shortcut.key + '|' + this.context, shortcut)

  handleKeyDown(event: KeyboardEvent): boolean {
    const key = this.formatKey(event)
    const shortcut = this.shortcuts.get(key + '|' + this.context)

    if (shortcut) {
      // 检查修饰键
      if (shortcut.ctrl && !event.ctrlKey) return false
      if (shortcut.alt && !event.altKey) return false
      if (shortcut.shift && !event.shiftKey) return false

      // 执行快捷键动作
      event.preventDefault()
      shortcut.action(event)
      return true
    }

    return false
  }

    return Array.from(this.shortcuts.values()).filter(
      s => s.context === context || s.context === 'global'
