```typescript
// src/lib/audio/__tests__/transcriber.test.ts
import { describe, it, expect, vi } from 'vitest'
import { AudioTranscriber } from '../transcriber'

describe('AudioTranscriber', () => {
  it('should transcribe audio correctly', async () => {
    const transcriber = new AudioTranscriber()
    const mockAudio = new Blob(['audio data'], { type: 'audio/wav' })

    const result = await transcriber.transcribe(mockAudio)

    expect(result.text).toBeDefined()
    expect(result.confidence).toBeGreaterThan(0.8)
    expect(result.duration).toBeGreaterThan(0)
  })

  it('should handle empty audio', async () => {
    const transcriber = new AudioTranscriber()
    const emptyAudio = new Blob([], { type: 'audio/wav' })

    await expect(transcriber.transcribe(emptyAudio))
      .rejects.toThrow('Empty audio file')
  })
})
```

```typescript
// src/lib/audio/__tests__/integration.test.ts
import { describe, it, expect, beforeAll, afterAll } from 'vitest'
import { setupServer } from 'msw/node'
import { rest } from 'msw'
import { AudioService } from '../service'

const server = setupServer(
  rest.post('/api/audio/transcribe', (req, res, ctx) => {
    return res(
      ctx.json({
        text: 'Hello world',
        confidence: 0.95,
        duration: 2.5,
      })
    )
  })
)

describe('AudioService Integration', () => {
  beforeAll(() => server.listen())
  afterAll(() => server.close())

  it('should upload and transcribe audio', async () => {
    const service = new AudioService()
    const audio = new Blob(['audio data'], { type: 'audio/wav' })

    const result = await service.uploadAndTranscribe(audio)

    expect(result.text).toBe('Hello world')
    expect(result.confidence).toBe(0.95)
  })
})
```

```typescript
import { test, expect } from '@playwright/test'

test.describe('Audio Processing', () => {

    await page.goto('/audio')

    // 上传音频文件
```
